* create `webpack.config.js`
* create `.babelrc`
* create dev envirement
```sh
npm init -y
npm add -D \
@babel/core \
@babel/preset-env \
@babel/preset-react \
@babel/plugin-proposal-class-properties \
@babel/plugin-proposal-object-rest-spread

npm add -D \
autoprefixer \
clean-webpack-plugin \
compression-webpack-plugin \
copy-webpack-plugin \
css-loader \
html-webpack-plugin \
node-sass \
postcss-easy-import \
postcss-loader \
sass-loader \
style-loader \
url-loader \
webpack \
webpack-cli

npm add -D babel-loader
```
